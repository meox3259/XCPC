# Berlekamp-Massey 

```cpp
/*
BM线性递推
接口:calculate(vector<T> sequence, long long n);
sequence:所求数列前[0,k-1]项，且sequence大小为k
n:返回该数列第n+1项
时间复杂度O(k^2logn)
*/
template <class T, T P>
class Berlekamp_Massey { // BM算法
	using V = vector<T>;

	T power(T x, T t) {
		T ret = 1;
		for(; t; t >>= 1, x = x * x % P) {
			if(t & 1) {
				ret = ret * x % P;
			}
		}
		return ret;
	}

	T inv(T x) {
		return power(x, P - 2);
	}

	public:
	V mult_mod_f(const V &x, const V &y, const V &f) {
    	int n = x.size();
		int m = y.size();
    	V res(n + m - 1);
    	for(int i = 0; i < n; ++i) {
        	for(int j = 0; j < m; ++j) {
            	res[i + j] = (res[i + j] + x[i] * y[j] % P) % P;
			}
		}
    	int k = f.size();
    	for(int i = n + m - 2; i >= k - 1; --i) {
       		T w = res[i] * inv(f[k - 1]) % P;
        	for(int p = 0; p < k - 1; ++p) {
            	res[i - (k - 1) + p] = (res[i - (k - 1) + p] - w * f[p] % P + P) % P;
			}
    	}
    	res.resize(k - 1);
    	return res;
	}

	V getf(const V &p) {
    	int sz = p.size();
    	V f(sz + 1);
    	f[sz] = 1;
    	for(int i = 0; i < sz; ++i) {
        	f[sz - 1 - i] = P - p[i];
		}
    	return f;
	}

	V getweight(const V &f, T q) {
    	int n = f.size();
    	V x(n);
    	x[0] = 1;
    	V z{ 0, 1 };
    	while(q) {
        	if(q & 1) {
            	x = mult_mod_f(x, z, f);
			}
        	z = mult_mod_f(z, z, f);
        	q >>= 1;
    	}
    	return x;
	}

	V BM(const V &s) {
    	V C(1, 1);
		V B(1, 1);
    	int L = 0;
		int m = 1;
		int b = 1;
    	for(int n = 0; n < s.size(); ++n) {
        	T d = 0;
        	for(int i = 0; i < L + 1; ++i) {
            	d = (d + C[i] * s[n - i]) % P;
			}
        	if(d == 0) {
            	++m;
			} else if (2 * L <= n) {
            	V tmp(C);
            	T c = P - d * power(b, P - 2) % P;
            	while(C.size() < B.size() + m) {
            	    C.push_back(0);
				}
            	for(int i = 0; i < B.size(); ++i) {
            	    C[i + m] = (C[i + m] + c * B[i]) % P;
				}
            	L = n + 1 - L;
            	B = tmp;
            	b = d;
            	m = 1;
        	} else {
            	T c = P - d * inv(b) % P;
            	while(C.size() < B.size() + m) {
            	    C.push_back(0);
				}
            	for(int i = 0; i < B.size(); ++i) {
            	    C[i + m] = (C[i + m] + c * B[i]) % P;
				}
            	++m;
        	}
    	}
    	C.erase(C.begin());
    	for(int i = 0; i < C.size(); ++i) {
        	C[i] = (P - C[i]) % P;
		}
    	return C;
	}

	T calculate(const V &a, long long n) {
   		V c = BM(a);       //求解递推 O(a^2)
    	V f = getf(c);     //构造f
    	V s = getweight(f, n); //求解系数 O(c^2logn)
    	T ret = 0;
    	for(int i = 0; i < s.size(); ++i) {
        	ret = (ret + s[i] * a[i] % P) % P;
		}
    	return ret;
	}
};
```
